int u;
typedef struct
list
{
	int n;
	struct list *p;
} *list;
list fun()
{
	int x;
	list h,t;
	t=(list)malloc(sizeof(list));
	h=(list)malloc(sizeof(list));
	t->p=NULL;
	h->p=NULL;
	scanf("%d",&x);
	while(x!=990122)
	{
		(*h).n=x;
		t=h;
		h=(list)malloc(sizeof(list));
		h->p=t;
		scanf("%d",&x);
		u++;
	}
	return t;
}
void gout(list l)
{
	list m;
	m=l;
    while(m->p)
	{
		printf("%d  ",m->n);
		m=m->p;
	}
	printf("%d  ",m->n);
}

	else

	   t=t->p;
	   m++;

	t->p=k;

}
void freedom(list l)
{
	list t;
	t=l;
	while(t->p)
	{
		free(t);
		l=l->p;
		t=l;
	}
	free(t);
}
